### **¬øQu√© es useEffect?**

`useEffect` es un hook que permite ejecutar c√≥digo con efectos secundarios en componentes funcionales. Es el equivalente a `componentDidMount`, `componentDidUpdate` y `componentWillUnmount` combinados.

#### **Casos de uso principales:**
- **Llamadas a APIs**: Obtener datos del servidor
- **Suscripciones**: Event listeners, timers, WebSockets
- **Sincronizaci√≥n**: Actualizar estado basado en props o estado
- **Efectos de limpieza**: Limpiar recursos cuando el componente se desmonta

### **3. Hook Personalizado para Efectos**

```javascript:src/hooks/useInterval.js
import { useEffect, useRef } from 'react';

// Hook personalizado para intervalos con useEffect
const useInterval = (callback, delay, options = {}) => {
  const {
    immediate = false, // Ejecutar inmediatamente
    pauseOnBlur = false, // Pausar cuando la app no est√° en foco
    maxExecutions = null, // N√∫mero m√°ximo de ejecuciones
  } = options;

  const savedCallback = useRef();
  const intervalRef = useRef(null);
  const executionCount = useRef(0);
  const isPaused = useRef(false);

  // Guardar callback en ref para evitar recrear el efecto
  useEffect(() => {
    savedCallback.current = callback;
  }, [callback]);

  // Efecto principal para el intervalo
  useEffect(() => {
    // Funci√≥n que se ejecutar√° en cada intervalo
    const tick = () => {
      if (isPaused.current) return;

      // Verificar l√≠mite de ejecuciones
      if (maxExecutions && executionCount.current >= maxExecutions) {
        clearInterval(intervalRef.current);
        return;
      }

      // Ejecutar callback
      if (savedCallback.current) {
        savedCallback.current();
        executionCount.current += 1;
      }
    };

    // Ejecutar inmediatamente si est√° habilitado
    if (immediate && delay !== null) {
      tick();
    }

    // Crear intervalo
    if (delay !== null) {
      intervalRef.current = setInterval(tick, delay);
    }

    // Funci√≥n de limpieza
    return () => {
      if (intervalRef.current) {
        clearInterval(intervalRef.current);
        intervalRef.current = null;
      }
    };
  }, [delay, immediate, maxExecutions]);

  // Efecto para manejar pausa en blur (opcional)
  useEffect(() => {
    if (!pauseOnBlur) return;

    const handleAppStateChange = (nextAppState) => {
      if (nextAppState === 'active') {
        isPaused.current = false;
      } else if (nextAppState === 'background' || nextAppState === 'inactive') {
        isPaused.current = true;
      }
    };

    // Aqu√≠ normalmente usar√≠as AppState de React Native
    // Para este ejemplo, simulamos con un listener personalizado
    
    return () => {
      // Limpiar listener si es necesario
    };
  }, [pauseOnBlur]);

  // Funci√≥n para pausar manualmente
  const pause = () => {
    isPaused.current = true;
  };

  // Funci√≥n para reanudar manualmente
  const resume = () => {
    isPaused.current = false;
  };

  // Funci√≥n para resetear contador de ejecuciones
  const reset = () => {
    executionCount.current = 0;
  };

  // Funci√≥n para obtener estad√≠sticas
  const getStats = () => ({
    executionCount: executionCount.current,
    isPaused: isPaused.current,
    isActive: intervalRef.current !== null,
    maxExecutions,
    hasReachedMax: maxExecutions ? executionCount.current >= maxExecutions : false,
  });

  return {
    pause,
    resume,
    reset,
    getStats,
  };
};

export default useInterval;
```
